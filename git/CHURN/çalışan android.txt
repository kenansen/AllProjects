// 🔹 API Base URL - Emulator için
private const val BASE_URL = "http://10.0.2.2:5000/"

// 🔹 API Request Model - YENİ PARAMETRELER EKLENDİ
data class PredictionRequest(
    @SerializedName("customerId") val customerId: String,
    @SerializedName("stockCode") val stockCode: String,
    @SerializedName("quantity") val quantity: String,
    @SerializedName("price") val price: String,
    @SerializedName("frequency") val frequency: String,

    // API tarafından beklenen eksik RFM ve Ülke parametreleri:
    @SerializedName("recency") val recency: String,         // Son alışverişten bu yana gün
    @SerializedName("tenure") val tenure: String,           // Müşteri olma süresi (gün)
    @SerializedName("numProducts") val numProducts: String, // Aldığı farklı ürün sayısı
    @SerializedName("country") val country: String,         // Ülke adı

    @SerializedName("model") val model: String
)

// 🔹 API Response Model
data class PredictionResponse(
    @SerializedName("success") val success: Boolean,
    @SerializedName("churnProbability") val churnProbability: Double,
    @SerializedName("model") val model: String,
    @SerializedName("message") val message: String? = null,
    @SerializedName("error") val error: String? = null
)

// 🔹 Retrofit API Interface
interface ChurnApiService {
    @POST("predict")
    suspend fun predictChurn(@Body request: PredictionRequest): PredictionResponse
}

// 🔹 Retrofit Instance
object RetrofitInstance {
    private val retrofit by lazy {
        Retrofit.Builder()
            .baseUrl(BASE_URL)
            .addConverterFactory(GsonConverterFactory.create())
            .build()
    }

    val api: ChurnApiService by lazy {
        retrofit.create(ChurnApiService::class.java)
    }
}

// 🔹 Müşteri verisi data class
data class CustomerRecord(
    val id: String,
    val name: String,
    val tenure: String,
    val contract: String,
    val paymentMethod: String,
    val monthlyCharges: String
)

// 🔹 Ekran enumları
private enum class Screen {
    MAIN_MENU,
    ADD_NEW_RECORD,
    DATASET_SELECTION,
    MODEL_SELECTION,
    CHURN_PREDICTION,
    PREDICTION_RESULT,
    RECORD_LIST
}

class MainActivity : ComponentActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContent {
            CustomerChurnAppTheme {
                Surface(
                    modifier = Modifier.fillMaxSize(),
                    color = MaterialTheme.colorScheme.background
                ) {
                    var currentScreen by remember { mutableStateOf(Screen.MAIN_MENU) }
                    var selectedDataset by remember { mutableStateOf<String?>(null) }
                    var selectedModel by remember { mutableStateOf<String?>(null) }
                    var predictionResult by remember { mutableStateOf(0) }
                    var customerRecords by remember { mutableStateOf(listOf<CustomerRecord>()) }
                    var csvRecords by remember { mutableStateOf(listOf<CustomerRecord>()) }

                    val context = LocalContext.current

                    // 🔹 CSV dosyasını yükle (ilk açılışta)
                    LaunchedEffect(Unit) {
                        try {
                            val inputStream = context.assets.open("customer_data.csv")
                            val reader = CSVReader(InputStreamReader(inputStream))
                            val allRows = reader.readAll()

                            // İlk satır header ise atla
                            val dataRows = allRows.drop(1)

                            csvRecords = dataRows.mapIndexed { index, row ->
                                CustomerRecord(
                                    id = index.toString(),
                                    name = row.getOrNull(0) ?: "",
                                    tenure = row.getOrNull(1) ?: "",
                                    contract = row.getOrNull(2) ?: "",
                                    paymentMethod = row.getOrNull(3) ?: "",
                                    monthlyCharges = row.getOrNull(4) ?: ""
                                )
                            }

                            reader.close()
                        } catch (e: Exception) {
                            println("CSV yükleme hatası: ${e.message}")
                        }
                    }

                    when (currentScreen) {
                        Screen.MAIN_MENU -> {
                            MainMenuScreen(
                                onNewRecord = {
                                    currentScreen = Screen.ADD_NEW_RECORD
                                },
                                onViewData = {
                                    currentScreen = Screen.DATASET_SELECTION
                                }
                            )
                        }

                        Screen.ADD_NEW_RECORD -> {
                            AddNewRecordScreen(
                                onSave = { record ->
                                    customerRecords = customerRecords + record
                                    currentScreen = Screen.RECORD_LIST
                                },
                                onBack = {
                                    currentScreen = Screen.MAIN_MENU
                                }
                            )
                        }

                        Screen.RECORD_LIST -> {
                            RecordListScreen(
                                records = customerRecords + csvRecords, // Manuel + CSV verileri birleşiyor
                                onBack = {
                                    currentScreen = Screen.MAIN_MENU
                                }
                            )
                        }

                        Screen.DATASET_SELECTION -> {
                            DatasetSelectionScreen(
                                onDatasetSelected = { datasetName ->
                                    selectedDataset = datasetName
                                },
                                onNext = {
                                    currentScreen = Screen.MODEL_SELECTION
                                },
                                onBack = {
                                    currentScreen = Screen.MAIN_MENU
                                }
                            )
                        }

                        Screen.MODEL_SELECTION -> {
                            ModelSelectionScreen(
                                onModelSelected = { modelName ->
                                    selectedModel = modelName
                                    currentScreen = Screen.CHURN_PREDICTION
                                },
                                onBack = {
                                    currentScreen = Screen.DATASET_SELECTION
                                }
                            )
                        }

                        Screen.CHURN_PREDICTION -> {
                            ChurnPredictionScreen(
                                selectedModel = selectedModel ?: "Random Forest",
                                onPredictionResult = { probability ->
                                    predictionResult = probability.toInt()
                                    currentScreen = Screen.PREDICTION_RESULT
                                },
                                onBack = {
                                    currentScreen = Screen.MODEL_SELECTION
                                }
                            )
                        }

                        Screen.PREDICTION_RESULT -> {
                            PredictionResultScreen(
                                churnProbability = predictionResult,
                                selectedModel = selectedModel ?: "Unknown",
                                onGoBack = {
                                    currentScreen = Screen.MAIN_MENU
                                }
                            )
                        }
                    }
                }
            }
        }
    }
}

@Composable
fun RecordListScreen(
    records: List<CustomerRecord>,
    onBack: () -> Unit
) {
    Column(
        modifier = Modifier
            .fillMaxSize()
            .padding(horizontal = 28.dp, vertical = 20.dp)
    ) {
        Text(
            text = "Müşteri Kayıtları",
            fontSize = 24.sp,
            fontWeight = FontWeight.Bold,
            color = Color(0xFF123E63),
            modifier = Modifier.align(Alignment.CenterHorizontally)
        )

        Spacer(modifier = Modifier.height(20.dp))

        if (records.isEmpty()) {
            Box(
                modifier = Modifier
                    .weight(1f)
                    .fillMaxWidth(),
                contentAlignment = Alignment.Center
            ) {
                Text(
                    text = "Henüz kayıt yok",
                    fontSize = 16.sp,
                    color = Color.Gray
                )
            }
        } else {
            LazyColumn(
                modifier = Modifier.weight(1f),
                verticalArrangement = Arrangement.spacedBy(12.dp)
            ) {
                items(records) { record ->
                    CustomerRecordCard(record)
                }
            }
        }

        Spacer(modifier = Modifier.height(12.dp))

        Button(
            onClick = onBack,
            modifier = Modifier
                .fillMaxWidth()
                .height(56.dp),
            shape = RoundedCornerShape(12.dp),
            colors = ButtonDefaults.buttonColors(containerColor = Color.LightGray)
        ) {
            Text("GERİ", fontSize = 16.sp, color = Color.Black)
        }
    }
}

        Button(
            onClick = {
                // Temel doğrulama: Zorunlu alanların kontrolü
                if (customerId.isNotBlank() && quantity.isNotBlank() && price.isNotBlank() && recency.isNotBlank() && tenure.isNotBlank()) {
                    isLoading = true
                    errorMessage = null

                    scope.launch {
                        try {
                            val request = PredictionRequest(
                                customerId = customerId,
                                stockCode = stockCode.ifBlank { "0" }, // Boşsa varsayılan
                                quantity = quantity,
                                price = price,
                                frequency = frequency.ifBlank { "1" }, // Boşsa varsayılan

                                // YENİ PARAMETRELERİN GÖNDERİLMESİ
                                recency = recency,
                                tenure = tenure,
                                numProducts = numProducts.ifBlank { "1" }, // Boşsa varsayılan
                                country = country.ifBlank { "United Kingdom" }, // Boşsa varsayılan

                                model = selectedModel
                            )

                            val response = RetrofitInstance.api.predictChurn(request)

                            isLoading = false

                            if (response.success) {
                                onPredictionResult(response.churnProbability)
                            } else {
                                errorMessage = response.error ?: "Tahmin başarısız"
                            }
                        } catch (e: Exception) {
                            isLoading = false
                            errorMessage = "Bağlantı hatası: ${e.message}"
                        }
                    }
                } else {
                    errorMessage = "Lütfen gerekli tüm alanları doldurun."
                }
            },
            modifier = Modifier
                .fillMaxWidth()
                .height(56.dp),
            shape = RoundedCornerShape(12.dp),
            colors = ButtonDefaults.buttonColors(containerColor = Color(0xFF123E63)),
            enabled = !isLoading
        ) {
            if (isLoading) {
                CircularProgressIndicator(color = Color.White, modifier = Modifier.size(24.dp))
            } else {
                Text("SUBMIT", fontSize = 18.sp, color = Color.White)
            }
        }

        Spacer(modifier = Modifier.height(12.dp))

        Button(
            onClick = onBack,
            modifier = Modifier
                .fillMaxWidth()
                .height(50.dp),
            shape = RoundedCornerShape(12.dp),
            colors = ButtonDefaults.buttonColors(containerColor = Color.LightGray)
        ) {
            Text("GERİ", fontSize = 16.sp, color = Color.Black)
        }
    }
}
